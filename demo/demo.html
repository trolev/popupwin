Popupwin v. 2.1b - всплывающие окна.

Быстрый старт.

Html:

<a href="#popupwin" id="a-popupwin">Ссылка</a>
<div id="popupwin">
    <a href="#?" class="close"></a>
    <!-- Содержимое -->
</div>

Css:

#popupwin {
    position: absolute;
    display: none;
}

Минимальный JavaScript:

$(document).ready(function(){
    $('#a-popupwin').popupwin();
});


 Demo

Html:

<a href="#?" id="a-popupwin">Ссылка</a>
<div id="popupwin">
    <a href="#?" class="close"></a>
    <!-- Содержимое -->
</div>

Css:

#popupwin {
    position: absolute;
    display: none;
}

Минимальный JavaScript:

$(document).ready(function(){
    $('#a-popupwin').popupwin({
        pwinBlock: '#popupwin'
    });
}); Demo

Пример с callbacks.

$(document).ready(function(){
    $('#a-popupwin').popupwin({
        callbackOpen: function(block, link){
            alert('open');
        },
        callbackClose: function(block, link){
            alert('close');
        }
    });
}); Demo

Сменим: фон, скорость анимации, прозрачность фона и кнопку закрытия окна.

$(document).ready(function(){
    $('#a-popupwin').popupwin({
        pwinClose: '.close',
        pwinBack: '#fff',
        pwinOpacity: 0.6,
        pwinSpeed: 200
    });
}); Demo

Все параметры (по умолчанию).

$(document).ready(function(){
    $('#a-popupwin').popupwin({
        pwinBlock: false,
        pwinClose: '.close',
        pwinBack: '#000', // При false фон не отобразится
        pwinOpacity: 0.3,
        pwinSpeed: 'slow',
        callbackOpen: function(block, link){},
        callbackClose: function(block, link){}
    });
});

И...

$(document).ready(function(){
    // Открыть блок #popupwin при загрузке страницы.
    $('#popupwin').popupwinOpen({
        pwinClose: '.close',
        pwinBack: '#000',
        pwinOpacity: 0.3,
        pwinSpeed: 'slow'
    });

    // Расположить блок #popupwin по центру экрана.
    $('#popupwin').popupwinCenter();
});
